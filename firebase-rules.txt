rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ---- AUTH HELPERS ----
    function isSignedIn() {
      return request.auth != null;
    }

    function getUserData() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }

    function isAdmin() {
      return isSignedIn() && 
        getUserData().role in ['admin', 'superadmin'];
    }

    function isStaff() {
      return isSignedIn() && 
        getUserData().role == 'staff';
    }

    // ---- USERS COLLECTION ----
    match /users/{userId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn();
      allow update: if isSignedIn() && (request.auth.uid == userId || isAdmin());
      allow delete: if false;
    }

    // ---- SERVICES COLLECTION ----
    match /services/{serviceId} {
      allow read: if isSignedIn();
      allow update: if isAdmin() || (isSignedIn() && (
        request.resource.data.keys().hasOnly(['projectCount', 'totalRating'])
      ));
      allow create, delete: if isAdmin();

      match /ratings/{ratingId} {
        allow create, read: if isSignedIn();
        allow update, delete: if false;
      }
    }

    // ---- BOOKINGS ----
    match /bookings/{bookingId} {
      // Permission for reading: 
      // 1. Client owns it
      // 2. Staff is assigned to it
      // 3. Admin/Superadmin
      // 4. Staff can see PENDING requests to accept them
      allow read: if isSignedIn() && (
        request.auth.uid == resource.data.userId || 
        request.auth.uid == resource.data.developerId || 
        isAdmin() ||
        (isStaff() && resource.data.status == 'PENDING') ||
        resource.data.status == 'COMPLETED'
      );

      allow create: if isSignedIn();

      allow update: if isSignedIn() && (
        request.auth.uid == resource.data.userId || 
        request.auth.uid == resource.data.developerId || 
        isAdmin()
      );

      allow delete: if isAdmin() || (isSignedIn() && request.auth.uid == resource.data.userId && resource.data.status == 'PENDING');
    }

    // ---- REQUESTS (LEGACY SUPPORT) ----
    match /requests/{requestId} {
      allow read, update: if isSignedIn() && (
        request.auth.uid == resource.data.user_id || 
        request.auth.uid == resource.data.assigned_to || 
        isAdmin()
      );
      allow create: if isSignedIn();
      allow delete: if isAdmin();
    }

    // ---- NOTIFICATIONS ----
    match /notifications/{notificationId} {
      allow read: if isSignedIn() && request.auth.uid == resource.data.userId;
      allow create: if isSignedIn();
      allow update: if isSignedIn() && request.auth.uid == resource.data.userId;
      allow delete: if false;
    }

    // ---- PAYMENTS & CONFIG ----
    match /payments/{paymentId} {
      allow read: if isSignedIn() && (request.auth.uid == resource.data.userId || isAdmin());
      allow create: if isSignedIn();
    }

    match /system_config/{docId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }
  }
}
